import bpy
C = bpy.context
name = C.active_object.material_slots[C.object.active_material_index].name
for o in C.scene.objects:
    if o.material_slots:
        for s in o.material_slots:
            if s.material:
                m = s.material
                if m.name == name:
                    print(o.name, [i for i, j in enumerate(o.layers) if j])
                    o.select = True
